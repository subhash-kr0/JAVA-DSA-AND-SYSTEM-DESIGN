2. you will close all the resources.

public static void cleanUp(Connection con ,PreparedStatement statement,ResultSet resultset) throws SQLException {
		//step 6 
	if(con!=null) {
		con.close();
	}
	if(statement != null) {
		statement.close();
	}
    if(resultset != null) {
    	resultset.close();
    }
	}
